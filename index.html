Project 5: Zero Hunger Maze Game

I created a maze game where the ball must reach the finish to represent defeating hunger. I used the onKeyHold function to make movement smooth and responsive. If you reach the goal without hitting walls — you win! This game symbolizes the journey toward Zero Hunger.

Label("Reduce Zero Hunger Game",200,60,size=20,bold=True)

walls = Group(
    Rect(55,0,65,150,fill='white'),

    Rect(270,85,65,180,fill='white'),
    Rect(0,265,335,65,fill='white'),
)
walls.visible = False

    walls.visible = True
    Rect(0,0,400,400,fill='lightGreen')

def onKeyHold(keys):
    c.toFront()
    if (c.hitsShape(walls) == False):
        app.stop()
    if ('up' in keys):
        c.centerY -= 10
    elif ('down' in keys):
        c.centerY += 10
    if ('right' in keys):
        c.centerX += 10
    elif ('left' in keys):
        c.centerX -= 10
    if -20 <= c.centerX <= 20 and 280 <= c.centerY <= 320:
        Label('You won!',200,200,size=30,bold=True,fill='green')
        app.stop()
