Project 5: Zero Hunger Maze Game

In this game, I created a ball that must reach the goal and fight against hunger! I used onKeyHold to move the circle through a maze. When it reaches the goal, it says "You won!" and ends the game. It's a cool interactive project combining logic and design.

app.background='yellow'
Label("Reduce Zero Hunger Game",200,60,size=20,bold=True)
Label("LETS GO!",185,110,size=25,bold=True,fill='blue')
Rect(60,220,280,120,fill=None,border='black',borderWidth=5)
l = Label("Press anywhere to start",200,280,size=22,bold=True)
c = Circle(85, 22, 20, fill='plum')

walls = Group(
    Rect(55,0,65,150,fill='white'),
    Rect(120,85,150,65,fill='white'),
    Rect(270,85,65,180,fill='white'),
    Rect(0,265,335,65,fill='white'),
)
walls.visible = False

def onMousePress(mouseX,mouseY):
    walls.visible = True
    Rect(0,0,400,400,fill='lightGreen')
    walls.toFront()
    c.toFront()

def onKeyHold(keys):
    c.toFront()
    if (c.hitsShape(walls) == False):
        app.stop()
    if ('up' in keys):
        c.centerY -= 10
    elif ('down' in keys):
        c.centerY += 10
    if ('right' in keys):
        c.centerX += 10
    elif ('left' in keys):
        c.centerX -= 10
    if -20 <= c.centerX <=20 and 280 <=c.centerY <=320:
        Label('You won!',200,200,size=30,bold=True,fill='green')
        app.stop()
